STEP 2: Create service-worker.js

const CACHE_NAME = 'funlearnbd-v1';
const urlsToCache = [
  './',
  './index.html',
  './images/numberblocks/1.png',
  './images/numberblocks/2.png',
  './images/numberblocks/3.png',
  './images/numberblocks/4.png',
  './images/numberblocks/5.png',
  './images/numberblocks/6.png',
  './images/numberblocks/7.png',
  './images/numberblocks/8.png',
  './images/numberblocks/9.png',
  './images/numberblocks/10.png'
];

self.addEventListener('install', event => {
  event.waitUntil(
    caches.open(CACHE_NAME)
      .then(cache => cache.addAll(urlsToCache))
  );
});

self.addEventListener('fetch', event => {
  event.respondWith(
    caches.match(event.request)
      .then(response => response || fetch(event.request))
  );
});
